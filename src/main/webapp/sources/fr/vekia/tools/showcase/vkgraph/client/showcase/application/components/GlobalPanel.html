30 public class GlobalPanel implements Display {
31 
32   private DeckLayoutPanel tabLayout;
33   private PresentationDemoPanel presentationPanel;
34   private Anchor returnLbl;
35 
36   private SimplePanel containerWidget;
37 
38   /**
39    * Default constructor
40    * 
41    */
42   public GlobalPanel() {
43     this.returnLbl = new Anchor("<< Retour");
44     this.returnLbl.setStyleName("vkl-Anchor");
45     this.containerWidget = new SimplePanel();
46     tabLayout = new DeckLayoutPanel();
47     tabLayout.setAnimationDuration(800);
48     tabLayout.addStyleName("vkl-CenterPanel");
49     presentationPanel = new PresentationDemoPanel();
50     this.setWidget(presentationPanel);
51   }
52 
53   @Override
54   public Widget asWidget() {
55     return tabLayout;
56   }
57 
58   /**
59      * 
60      */
61   public void removeItem(IsWidget item) {
62     tabLayout.remove((Widget) item);
63   }
64 
65   /**
66    * @return the returnLbl
67    */
68   @Override
69   public HasClickHandlers getReturnLbl() {
70     return returnLbl;
71   }
72 
73   @Override
74   public HasClickHandlers getStartAutomatedDemoHandler() {
75     return presentationPanel.getStartAutomatedDemoHandler();
76   }
77 
78   @Override
79   public void setWidget(IsWidget w) {
80     this.containerWidget.setWidget(w);
81     reinitPanel();
82   }
83 
84   private final void reinitPanel() {
85     VerticalPanel p = new VerticalPanel();
86     p.setSize("100%", "100%");
87 
88     p.add(returnLbl);
89     p.setCellHeight(returnLbl, "10px");
90 
91     p.add(this.containerWidget);
92     this.containerWidget.setSize("100%", "100%");
93 
94     tabLayout.setWidget(p);
95     p.getElement().getParentElement().getStyle()
96         .setOverflow(Overflow.SCROLL);
97   }
