029 public class JQueryDialog extends Widget implements ResizeHandler,
030     HasResizeHandlers {
031 
032   private SimplePanel p;
033   private String id;
034   private String idContent;
035 
036   /**
037    * Default constructor
038    * 
039    */
040   public JQueryDialog(boolean isResizable, boolean isDraggable) {
041     this.p = new SimplePanel();
042     this.p.getElement().setId(DOM.createUniqueId());
043     this.id = this.p.getElement().getId();
044     int height = Window.getClientHeight() / 2;
045     int width = Window.getClientWidth() / 3;
046 
047     int x = 0;
048     int y = Window.getClientWidth() - width;
049     super.setElement(p.getElement());
050     super.setTitle("The Dialog");
051 
052     RootLayoutPanel.get().add(p);
053 
054     this.create(id, height, width, x, y, isResizable, isDraggable);
055 
056     this.addResizeHandler(this);
057   }
058 
059   /**
060  * 
061  */
062   private native void create(String id, int height, int width, int x, int y,
063       boolean isResizable, boolean isDraggable) /*-{
064                             $wnd.jQuery("#" + id).dialog({
065                             autoOpen : false,
066                             show : "blind",
067                             hide : "explode",
068                             height : height,
069                             width : width,
070                             draggable : isDraggable,
071                             resizable : isResizable,
072                             position : [ x, y ]
073                             });
074                             }-*/;
075 
076   /**
077      * 
078      */
079   public void show() {
080     this.open(id);
081   };
082 
083   /**
084      * 
085      */
086   private native void open(String id) /*-{
087                     $wnd.jQuery("#" + id).dialog("open");
088                     }-*/;
089 
090   @Override
091   public void onResize(ResizeEvent event) {
092     int height = DOM.getElementById(id).getOffsetHeight();
093     int width = DOM.getElementById(id).getOffsetWidth();
094 
095     DOM.getElementById(idContent).getStyle()
096         .setPropertyPx("maxHeight", height);
097     DOM.getElementById(idContent).getStyle()
098         .setPropertyPx("maxWidth", width);
099   }
100 
101   @Override
102   public HandlerRegistration addResizeHandler(ResizeHandler handler) {
103     return addHandler(handler, ResizeEvent.getType());
104   }
105 
106   /**
107    * @param bar
108    */
109   public void setWidget(Widget widget) {
110     if (widget.getElement().getId() == null) {
111       widget.getElement().setId(DOM.createUniqueId());
112     }
113     idContent = widget.getElement().getId();
114     p.setWidget(widget);
115   }
