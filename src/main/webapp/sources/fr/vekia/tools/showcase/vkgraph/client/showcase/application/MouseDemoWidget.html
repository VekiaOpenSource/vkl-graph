26 public final class MouseDemoWidget extends Animation {
27   private static MouseDemoWidget instance = null;
28   private static final Image img = new Image("img/mouse.png");
29 
30   private final Element element;
31   private int startX;
32   private int startY;
33   private int finalX;
34   private int finalY;
35 
36   /**
37    * Default constructor
38    * 
39    */
40   private MouseDemoWidget() {
41     element = img.getElement();
42 
43   }
44 
45   public static MouseDemoWidget getInstance() {
46     if (instance == null) {
47       instance = new MouseDemoWidget();
48     }
49     return instance;
50   }
51 
52   public void scrollTo(int x, int y, int milliseconds) {
53     this.finalX = x;
54     this.finalY = y;
55 
56     startX = element.getOffsetLeft();
57     startY = element.getOffsetTop();
58 
59     run(milliseconds);
60   }
61 
62   public void scrollTo(UIObject object, int milliseconds) {
63     if (object.isVisible()) {
64       scrollTo(object.getAbsoluteLeft() + 10,
65           object.getAbsoluteTop() + 5, milliseconds);
66     }
67   }
68 
69   @Override
70   protected void onUpdate(double progress) {
71     double positionX = startX + (progress * (this.finalX - startX));
72     double positionY = startY + (progress * (this.finalY - startY));
73 
74     this.element.getStyle().setLeft(positionX, Style.Unit.PX);
75     this.element.getStyle().setTop(positionY, Style.Unit.PX);
76   }
77 
78   @Override
79   protected void onComplete() {
80     super.onComplete();
81     this.element.getStyle().setLeft(this.finalX, Style.Unit.PX);
82     this.element.getStyle().setTop(this.finalY, Style.Unit.PX);
83   }
84 
85   public void activate() {
86     RootLayoutPanel.get().add(img);
87   }
88 
89   public void desactivate() {
90     RootLayoutPanel.get().remove(img);
91 
92   }
