010 package fr.vekia.tools.showcase.vkgraph.client.showcase.application.components;
011 
012 import java.util.HashMap;
013 import java.util.Map;
014 
015 import com.google.gwt.dom.client.Style.Unit;
016 import com.google.gwt.safehtml.shared.SafeHtml;
017 import com.google.gwt.user.client.ui.StackLayoutPanel;
018 import com.google.gwt.user.client.ui.TreeItem;
019 import com.google.gwt.user.client.ui.UIObject;
020 import com.google.gwt.user.client.ui.Widget;
021 
022 import fr.vekia.tools.showcase.vkgraph.client.showcase.application.ShowcaseMenuPresenter.Display;
023 import fr.vekia.tools.showcase.vkgraph.client.showcase.application.components.menu.AbstractShowcaseTreeMenuItem;
024 import fr.vekia.tools.showcase.vkgraph.client.showcase.application.components.menu.ShowcaseMenuTree;
025 
026 /**
027  * @author Steeve Vandecappelle (SVA)
028  * @since 23 avr. 2012. GWTQuery Vekia Showcase
029  * @version 1.0
030  * 
031  *          {@inheritDoc}
032  */
033 public class Menu implements Display {
034 
035   private StackLayoutPanel stackLayoutPanel;
036   private Map<String, ShowcaseMenuTree> menuTrees;
037 
038   /**
039    * Default constructor
040    * 
041    */
042   public Menu() {
043     stackLayoutPanel = new StackLayoutPanel(Unit.PX);
044     stackLayoutPanel.addStyleName("Showcase-Menu");
045     menuTrees = new HashMap<String, ShowcaseMenuTree>();
046     stackLayoutPanel.setHeight("100%");
047   }
048 
049   @Override
050   public int addElementWithHeader(AbstractShowcaseTreeMenuItem items,
051       SafeHtml header, int headerSize) {
052     ShowcaseMenuTree menuTree = new ShowcaseMenuTree(items);
053     stackLayoutPanel.add(menuTree, header, headerSize);
054     menuTrees.put(header.asString(), menuTree);
055     return stackLayoutPanel.getWidgetIndex(menuTree);
056   }
057 
058   @Override
059   public Widget asWidget() {
060     return stackLayoutPanel;
061   }
062 
063   @Override
064   public ShowcaseMenuTree getSelectionHandler(String headerText) {
065     return menuTrees.get(headerText);
066   }
067 
068   @Override
069   public void showWidget(int index) {
070     stackLayoutPanel.showWidget(index);
071   }
072 
073   @Override
074   public void selectTreeItem(String menuTitle,
075       AbstractShowcaseTreeMenuItem item) {
076     ShowcaseMenuTree menuTree = menuTrees.get(menuTitle);
077     menuTree.getTreeWidget().setSelectedItem(item);
078   }
079 
080   @Override
081   public void addHeaderStyleName(String style, Integer index) {
082     stackLayoutPanel.getHeaderWidget(index).addStyleName(style);
083   }
084 
085   @Override
086   public void removeHeaderStyleName(String style, Integer index) {
087     stackLayoutPanel.getHeaderWidget(index).removeStyleName(style);
088   }
089 
090   @Override
091   public void addItemStyleName(String menuTitle, String itemText, String style) {
092     getTreeItem(menuTitle, itemText).addStyleName(style);
093   }
094 
095   @Override
096   public UIObject getHeaderWidget(int index) {
097     return stackLayoutPanel.getHeaderWidget(index);
098   }
099 
100   @Override
101   public UIObject getTreeItem(String menuTitle, String itemText) {
102     ShowcaseMenuTree menuTree = menuTrees.get(menuTitle);
103     return getTreeItem(menuTree.getItem(), menuTitle, itemText);
104   }
105 
106   private UIObject getTreeItem(TreeItem treeItem, String menuTitle,
107       String itemText) {
108     for (int i = 0; i < treeItem.getChildCount(); i++) {
109       TreeItem treeItemChild = treeItem.getChild(i);
110       if (treeItemChild.getText().equals(itemText)) {
111         return treeItemChild;
112       } else if (treeItemChild.getChildCount() > 0) {
113         return getTreeItem(treeItemChild, menuTitle, itemText);
114       }
115     }
116     return treeItem;
117   }
118 
119   @Override
120   public void removeItemStyleName(String menuTitle, String itemText,
121       String style) {
122     getTreeItem(menuTitle, itemText).removeStyleName(style);
123   }
124 }
