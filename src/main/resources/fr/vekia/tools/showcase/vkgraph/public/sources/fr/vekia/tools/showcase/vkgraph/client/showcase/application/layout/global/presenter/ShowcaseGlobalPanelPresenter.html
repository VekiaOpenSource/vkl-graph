023 /**
024  * @author Steeve Vandecappelle (SVA)
025  * @since 23 avr. 2012. GWTQuery Vekia Showcase
026  * @version 1.0
027  */
028 @Singleton
029 public class ShowcaseGlobalPanelPresenter extends WidgetPresenter<GlobalDisplay> {
030 
031     /**
032      * Default constructor
033      * 
034      * @param display
035      * @param eventBus
036      */
037     @Inject
038     public ShowcaseGlobalPanelPresenter(GlobalDisplay display, EventBus eventBus) {
039         super(display, eventBus);
040 
041         display.getReturnLbl().addClickHandler(new ClickHandler() {
042 
043             @Override
044             public void onClick(ClickEvent event) {
045                 History.back();
046             }
047         });
048     }
049 
050     @Override
051     protected void onBind() {
052         registerHandler(display.getStartAutomatedDemoHandler().addClickHandler(new ClickHandler() {
053 
054             @Override
055             public void onClick(ClickEvent event) {
056                 ShowcaseInjector.Util.getInstance().getEventBus().fireEvent(new StartAutomatedDemoEvent());
057             }
058         }));
059 
060         registerHandler(eventBus.addHandler(SelectionEvent.getType(), new SelectionHandler<AbstractShowcaseTreeMenuItem>() {
061 
062             @Override
063             public void onSelection(final SelectionEvent<AbstractShowcaseTreeMenuItem> event) {
064                 historyHandle(event.getSelectedItem());
065                 displayScreen(event.getSelectedItem());
066             }
067 
068         }));
069 
070     }
071 
072     private void historyHandle(AbstractShowcaseTreeMenuItem selectedItem) {
073 
074         if (selectedItem.hasScreen()) {
075             String categ = "";
076 
077             String screen = selectedItem.getText();
078 
079             while (selectedItem.getParentItem() != null) {
080                 categ = selectedItem.getParentItem().getText();
081                 selectedItem = (AbstractShowcaseTreeMenuItem) selectedItem.getParentItem();
082             }
083             // create the screen and attach to DOM
084             ShowcaseInjector.Util.getInstance().getPresenter().getPlaceController().goTo(new MenuPlace(categ, screen));
085         }
086 
087     }
088 
089     private final void displayScreen(final AbstractShowcaseTreeMenuItem abstractShowcaseTreeMenuItem) {
090         Scheduler.get().scheduleFinally(new ScheduledCommand() {
091 
092             @Override
093             public void execute() {
094 
095                 final IsWidget code = abstractShowcaseTreeMenuItem.getScreen();
096                 eventBus.fireEvent(new SelectionCodeEvent(code));
097             }
098         });
099 
100     }
101 
102     @Override
103     protected void onUnbind() {
104     }
105 
106     @Override
107     protected void onRevealDisplay() {
108     }
109 }
