001 package fr.vekia.tools.showcase.vkgraph.client.showcase.application.components;
002 
003 import com.google.gwt.event.dom.client.ContextMenuEvent;
004 import com.google.gwt.event.dom.client.ContextMenuHandler;
005 import com.google.gwt.event.shared.GwtEvent;
006 import com.google.gwt.event.shared.HandlerRegistration;
007 import com.google.gwt.user.client.Command;
008 import com.google.gwt.user.client.ui.IsWidget;
009 import com.google.gwt.user.client.ui.Label;
010 import com.google.gwt.user.client.ui.MenuBar;
011 import com.google.gwt.user.client.ui.MenuItem;
012 import com.google.gwt.user.client.ui.PopupPanel;
013 import com.google.gwt.user.client.ui.Widget;
014 
015 import fr.vekia.tools.showcase.vkgraph.client.showcase.application.components.events.HasItemTabSelectionHandlers;
016 import fr.vekia.tools.showcase.vkgraph.client.showcase.application.components.events.ItemTabSelectionEvent;
017 import fr.vekia.tools.showcase.vkgraph.client.showcase.application.components.events.ItemTabSelectionHandler;
018 
019 /**
020  * @author Steeve Vandecappelle (SVA)
021  * @since 25 avr. 2012. GWTQuery Vekia Showcase
022  * @version 1.0
023  */
024 public class TabRemovableItem implements IsWidget, HasItemTabSelectionHandlers {
025 
026     private Label label;
027     private MenuTabRightClick menu;
028 
029     private class MenuTabRightClick extends PopupPanel {
030 
031         private MenuBar bar;
032 
033         /**
034          * Default constructor
035          * 
036          */
037         public MenuTabRightClick() {
038             super(true, false);
039             bar = new MenuBar(true);
040             this.add(bar);
041             this.bar.addItem(new MenuItem("Remove", new Command() {
042 
043                 @Override
044                 public void execute() {
045                     label.fireEvent(new ItemTabSelectionEvent(ItemActionSelection.DELETE));
046                     hide();
047                 }
048             }));
049             this.bar.addItem(new MenuItem("Remove other items", new Command() {
050 
051                 @Override
052                 public void execute() {
053                     label.fireEvent(new ItemTabSelectionEvent(ItemActionSelection.DELETE_OTHERS));
054                     hide();
055                 }
056             }));
057             this.bar.addItem(new MenuItem("Remove items on the right", new Command() {
058 
059                 @Override
060                 public void execute() {
061                     label.fireEvent(new ItemTabSelectionEvent(ItemActionSelection.DELETE_ONRIGHT));
062                     hide();
063                 }
064             }));
065             this.setAnimationEnabled(true);
066         }
067     }
068 
069     /**
070      * Default constructor
071      * 
072      * @param indexTab
073      * 
074      */
075     public TabRemovableItem(String itemText) {
076         label = new Label(itemText);
077         menu = new MenuTabRightClick();
078         label.addDomHandler(new ContextMenuHandler() {
079 
080             @Override
081             public void onContextMenu(ContextMenuEvent event) {
082                 event.preventDefault();
083 
084                 menu.showRelativeTo(label);
085             }
086         }, ContextMenuEvent.getType());
087     }
088 
089     @Override
090     public Widget asWidget() {
091         return label;
092     }
093 
094     @Override
095     public void fireEvent(GwtEvent<?> event) {
096         this.label.fireEvent(event);
097     }
098 
099     @Override
100     public HandlerRegistration addSelectionHandler(ItemTabSelectionHandler handler) {
101         return label.addHandler(handler, ItemTabSelectionEvent.getType());
102     }
103 }
