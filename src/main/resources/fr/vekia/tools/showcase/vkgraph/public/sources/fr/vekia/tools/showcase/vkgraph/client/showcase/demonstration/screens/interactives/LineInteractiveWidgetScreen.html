003 import java.util.ArrayList;
004 import java.util.List;
005 
006 import com.google.gwt.user.client.ui.FlexTable;
007 import com.google.gwt.user.client.ui.HorizontalPanel;
008 import com.google.gwt.user.client.ui.Label;
009 
010 import fr.vekia.vkgraph.client.charts.LineChart;
011 import fr.vekia.vkgraph.client.charts.events.DataClickEvent;
012 import fr.vekia.vkgraph.client.charts.events.EditedDataEvent;
013 import fr.vekia.vkgraph.client.charts.events.EventObject;
014 import fr.vekia.vkgraph.client.datas.OptionSerie;
015 import fr.vekia.vkgraph.client.datas.SeriesData;
016 import fr.vekia.vkgraph.client.options.ChartOption;
017 import fr.vekia.vkgraph.client.options.SubOption;
018 
019 /**
020  * @author Steeve Vandecappelle (SVA)
021  * @since 17 août 2012. VklGraph version 1.2
022  * @version 2.1
023  */
024 public class LineInteractiveWidgetScreen extends HorizontalPanel {
025     private Label eventTypeLabel;
026     private Label eventDataLabelSerie;
027     private Label eventDataLabelPoint;
028     private Label eventDataLabelValue;
029 
030     /**
031      * Default constructor
032      * 
033      */
034     public LineInteractiveWidgetScreen() {
035         eventTypeLabel = new Label();
036         eventDataLabelSerie = new Label();
037         eventDataLabelPoint = new Label();
038         eventDataLabelValue = new Label();
039 
040         LineChart<Number> chart = new LineChart<Number>();
041 
042         List<Number> datas = new ArrayList<Number>();
043         datas.add(5);
044         datas.add(17);
045         datas.add(9);
046         datas.add(11);
047         datas.add(10);
048         datas.add(15);
049         datas.add(17);
050         datas.add(14);
051         datas.add(12);
052         datas.add(10);
053         datas.add(8);
054         datas.add(12);
055         chart.setListView(true);
056         chart.setDatas(datas);
057 
058         chart.bind(new DataClickEvent() {
059 
060             @Override
061             public void onEvent(EventObject datasOnEvent) {
062                 eventTypeLabel.setText("click");
063                 eventDataLabelSerie.setText("Serie: " + datasOnEvent.getSeriesIndex());
064                 eventDataLabelPoint.setText("Point index: " + datasOnEvent.getPointIndex());
065                 eventDataLabelValue.setText("value:" + datasOnEvent.getValue());
066             }
067         });
068 
069         chart.bind(new EditedDataEvent() {
070 
071             @Override
072             public void onEvent(EventObject datasOnEvent) {
073                 eventTypeLabel.setText("drag");
074                 eventDataLabelSerie.setText("Serie: " + datasOnEvent.getSeriesIndex());
075                 eventDataLabelPoint.setText("Point index: " + datasOnEvent.getPointIndex());
076                 eventDataLabelValue.setText("value:" + datasOnEvent.getValue());
077             }
078         });
079 
080         chart.setTextOption(ChartOption.seriesDefaults, SubOption.dragable, SubOption.constrainTo, "y");
081         chart.setBooleanOption(ChartOption.animate, true);
082         chart.setBooleanOption(ChartOption.seriesDefaults, SubOption.rendererOptions, SubOption.smooth, true);
083 
084         chart.setPluginsEnable(true);
085 
086         add(chart);
087         FlexTable dataTable = new FlexTable();
088         dataTable.setWidth("100%");
089         add(dataTable);
090         setCellWidth(dataTable, "100%");
091         dataTable.getRowFormatter().setStyleName(0, "gwt-DataTableTitle");
092         dataTable.setWidget(0, 0, new Label("Event fired: "));
093         dataTable.setWidget(0, 1, eventTypeLabel);
094         dataTable.setWidget(0, 2, new Label("Event data: "));
095         dataTable.setWidget(0, 3, eventDataLabelSerie);
096         dataTable.setWidget(1, 3, eventDataLabelPoint);
097         dataTable.setWidget(2, 3, eventDataLabelValue);
098 
099         eventTypeLabel.setStylePrimaryName("gwt-ImportantLabel");
100         eventDataLabelSerie.setStylePrimaryName("gwt-ImportantLabel");
101         eventDataLabelPoint.setStylePrimaryName("gwt-ImportantLabel");
102         eventDataLabelValue.setStylePrimaryName("gwt-ImportantLabel");
103 
104         dataTable.addStyleName("gwt-DataTable");
105 
106         dataTable.getFlexCellFormatter().setColSpan(0, 0, 4);
107 
108         SeriesData dataserie = new SeriesData();
109         OptionSerie serie1 = new OptionSerie();
110 
111         serie1.setTextOption(SubOption.dragable, SubOption.constrainTo, "y");
112         serie1.setBooleanOption(SubOption.showMarker, false);
113 
114         dataserie.set(0, serie1);
115 
116         chart.setSeriesOptions(dataserie);
117 
118         chart.setResizable(true);
119     }
120 }
