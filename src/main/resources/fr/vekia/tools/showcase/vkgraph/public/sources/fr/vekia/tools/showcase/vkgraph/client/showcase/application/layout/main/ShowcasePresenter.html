021 /**

023  * @since 23 avr. 2012. GWTQuery Vekia Showcase

026 @Singleton
027 public class ShowcasePresenter extends WidgetPresenter<ShowcaseMainLayoutDisplay>implements ActivityPresenter {
028     private PlaceController controller;
029 
030     /**
031      * Default constructor
032      * 
033      * @param display
034      * @param eventBus
035      */
036     @Inject
037     public ShowcasePresenter(ShowcaseMainLayoutDisplay display, EventBus eventBus) {
038         super(display, eventBus);
039 
040         ShowcaseInjector injector = ShowcaseInjector.Util.getInstance();
041         ShowcaseMenuPresenter menuPresenter = injector.getMenuPresenter();
042         display.addWest(menuPresenter.getDisplay().asWidget(), 320);
043 
044         ShowcaseConsoleCodePresenter consoleCodePresenter = injector.getConsoleCodePresenter();
045         consoleCodePresenter.getDisplay().insert();
046 
047         ShowcaseGlobalPanelPresenter globalPanelPresenter = injector.getGlobalPanelPresenter();
048         display.addCenter(globalPanelPresenter.getDisplay().asWidget());
049 
050         menuPresenter.bind();
051         globalPanelPresenter.bind();
052         consoleCodePresenter.bind();
053     }
054 
055     @Override
056     protected void onBind() {
057         eventBus.addHandler(StartAutomatedDemoEvent.getType(), new StartAutomatedDemoHandler() {
058 
059             @Override
060             public void onStart(StartAutomatedDemoEvent event) {
061                 DemoStarter.start();
062             }
063         });
064     }
065 
066     @Override
067     protected void onUnbind() {
068     }
069 
070     @Override
071     protected void onRevealDisplay() {
072         RootLayoutPanel.get().insert(display.asWidget(), 0);
073     }
074 
075     /**
076      * @return the controller
077      */
078     public PlaceController getPlaceController() {
079         return controller;
080     }
081 
082     /**
083      * @param controller
084      *            the controller to set
085      */
086     public void setController(PlaceController controller) {
087         this.controller = controller;
088     }
089 
090     @Override
091     public void gotTo(MenuPlace menuPlace) {
092         controller.goTo(menuPlace);
093     }
094 
095     /**
096      * 
097      */
098     public void detachDisplay() {
099         RootLayoutPanel.get().remove(display.asWidget());
100     }
101 }
