05

08 import com.google.gwt.dom.client.Style.Unit;
09 import com.google.gwt.user.client.DOM;
10 import com.google.gwt.user.client.ui.Composite;
11 import com.google.gwt.user.client.ui.FlowPanel;
12 import com.google.gwt.user.client.ui.ScrollPanel;
13 
14 /**
15  * @author Steeve Vandecappelle (SVA)

19 public class CodeLine extends Composite {
20 
21     private String lineOfCode;
22     private FlowPanel code;
23 
24     /**
25      * Default constructor
26      * 
27      */
28     public CodeLine(String codeLine) {
29         this.lineOfCode = codeLine;
30 
31         addColors();
32         ScrollPanel panel = new ScrollPanel();
33         panel.setHeight("800px");
34         panel.add(code);
35         initWidget(panel);
36     }
37 
38     /**
39      * 
40      */
41     private void addColors() {
42         code = new FlowPanel();
43 
44         String html = lineOfCode;
45         html = html.replaceAll("<", "&#60;");
46         html = html.replaceAll("<", "&#62;");
47 
48         Element preElement = DOM.createElement("pre");
49 
50         List<String> keyWords = new ArrayList<String>();
51         keyWords.add("return");
52         keyWords.add("public");
53         keyWords.add("private");
54         keyWords.add("protected");
55         keyWords.add("package");
56         keyWords.add("void");
57         keyWords.add("boolean");
58         keyWords.add("null");
59         keyWords.add("this");
60         keyWords.add("new");
61         keyWords.add("class");
62         keyWords.add("extends");
63         keyWords.add("implements");
64         keyWords.add("super");
65         keyWords.add("import");
66         keyWords.add("for");
67         keyWords.add("while");
68         keyWords.add("final");
69         keyWords.add("static");
70 
71         for (String keyWord : keyWords) {
72             html = html.replaceAll("(" + keyWord + ")", "<span style=\"color: #B858F4; font-weight:bold;\">$1</span>");
73         }
74         html = html.replaceAll("(/\\*[.\\r\\s \\*\\w\\:\\$\\{\\}\\@\\(\\)]*/)", "<span style=\"color: #B98064;\">$1</span>");
75         html = html.replaceAll("([ ][A-Z]{1}[a-zA-Z]+)", "<span style=\"color: #81BD81;\">$1</span>");
76 
77         html = html.replaceAll("(^)", "$1<li class=\"alt\">");
78         html = html.replaceAll("($)", "</li>$1");
79 
80         preElement.setInnerHTML(html);
81         preElement.getStyle().setFontWeight(FontWeight.NORMAL);
82         preElement.getStyle().setFontSize(12, Unit.PT);
83         code.getElement().appendChild(preElement);
84     }
