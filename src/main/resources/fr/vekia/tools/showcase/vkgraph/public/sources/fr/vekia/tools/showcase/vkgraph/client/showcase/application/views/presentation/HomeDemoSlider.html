024 /**
025  * Image slider demo for home page.
026  * 
027  * @author svandecappelle
028  * @version 5.0.0
029  */
030 public class HomeDemoSlider extends SimplePanel {
031 
032     interface ImgRessource extends ClientBundle {
033         @ImageOptions(width = 1000, height = 650)
034         ImageResource screenshotOne();
035 
036         @ImageOptions(width = 1000, height = 650)
037         ImageResource screenshotTwo();
038 
039         final static class Util {
040             private static ImgRessource instance;
041 
042             /**
043              * Default constructor
044              * 
045              */
046             private Util() {
047             }
048 
049             static final ImgRessource getInstance() {
050                 if (instance == null) {
051                     instance = GWT.create(ImgRessource.class);
052                 }
053                 return instance;
054             }
055         }
056     }
057 
058     public HomeDemoSlider() {
059 
060         CallbackEventController<ConsoleSlideshowCallBack, IsWidget> callBackEventController = new CallbackEventController<ConsoleSlideshowCallBack, IsWidget>() {
061 
062             @Override
063             public void onSuccess(List<IsWidget> result) {
064                 final Slideshow slideshow = new Slideshow(Window.getClientWidth() / 2, Window.getClientHeight() / 2);
065                 Image screenshotOne = new Image(ImgRessource.Util.getInstance().screenshotOne());
066                 Image screenshotTwo = new Image(ImgRessource.Util.getInstance().screenshotTwo());
067                 Window.addResizeHandler(new ResizeHandler() {
068 
069                     @Override
070                     public void onResize(ResizeEvent event) {
071                         slideshow.setSize(Window.getClientWidth() / 2 + "px", Window.getClientHeight() / 2 + "px");
072                         slideshow.setSizeContainerPx(Window.getClientWidth() / 2, Window.getClientHeight() / 2);
073                         slideshow.redraw();
074                     }
075                 });
076                 slideshow.add(screenshotOne);
077                 slideshow.add(screenshotTwo);
078 
079                 for (IsWidget isWidget : result) {
080                     slideshow.add((Widget) isWidget);
081                 }
082 
083                 setWidget(slideshow);
084             }
085 
086             @Override
087             public void onFailure(Throwable caught) {
088                 GWT.log("not able to retreive code due to: ", caught);
089             }
090         };
091         final ConsoleSlideshowCallBack generalPlot = new ConsoleSlideshowCallBack(callBackEventController);
092         final ConsoleSlideshowCallBack areaPlot = new ConsoleSlideshowCallBack(callBackEventController);
093         final ConsoleSlideshowCallBack pyramidPlot = new ConsoleSlideshowCallBack(callBackEventController);
094         callBackEventController.addTraitment(new CodingGetShowcaseSlideshowCommand(generalPlot, SimplePlotScreen.class));
095         callBackEventController.addTraitment(new CodingGetShowcaseSlideshowCommand(areaPlot, AreaChartInteractiveWidgetScreen.class));
096         callBackEventController.addTraitment(new CodingGetShowcaseSlideshowCommand(pyramidPlot, PyramidChartInteractiveWidget.class));
097 
098         callBackEventController.execute();
099     }
100 
101     static class CodingGetShowcaseSlideshowCommand implements Command {
102 
103         private Class<?> classBinded;
104         private ConsoleSlideshowCallBack callBack;
105 
106         /**
107          * Default constructor
108          * 
109          */
110         public CodingGetShowcaseSlideshowCommand(ConsoleSlideshowCallBack callBack, Class<?> classBinded) {
111             this.classBinded = classBinded;
112             this.callBack = callBack;
113         }
114 
115         @Override
116         public void execute() {
117             CodeServiceUtil.getCode(this.classBinded.getName(), callBack);
118         }
119 
120     }
121 }
