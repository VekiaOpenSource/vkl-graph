084   /**

090   @Inject
091   public ShowcaseConsoleCodePresenter(Display display, EventBus eventBus) {
092     super(display, eventBus);
093 
094     this.display.getAddShowHandler().addClickHandler(new ClickHandler() {
095 
096       @Override
097       public void onClick(ClickEvent event) {
098         ShowcaseConsoleCodePresenter.this.display.show();
099       }
100     });
101   }
102 
103   public void showConsole() {
104     ShowcaseConsoleCodePresenter.this.display.show();
105   }
106 
107   public void showConsoleOutWindow() {
108     ShowcaseConsoleCodePresenter.this.display.showOutBrowser();
109   }
110 
111   @Override
112   protected void onBind() {
113     eventBus.addHandler(SelectionCodeEvent.getType(),
114         new SelectionCodeHandler() {
115 
116           @Override
117           public void onSelection(SelectionCodeEvent event) {
118 
119             CodeServiceUtil.getCode(event.getSelectedItem()
120                 .getClass().getName(),
121                 new AsyncCallback<String>() {
122 
123                   @Override
124                   public void onSuccess(String result) {
125                     display.setCode(result);
126                   }
127 
128                   @Override
129                   public void onFailure(Throwable caught) {
130                     GWT.log("not able to retreive code due to: ",
131                         caught);
132                     display.setCode("no code available");
133                     consolejs(caught.getMessage());
134                   }
135                 });
136 
137             CodeServiceUtil.getCodeUrl(event.getSelectedItem()
138                 .getClass().getName(),
139                 new AsyncCallback<CodeUrl>() {
140 
141                   @Override
142                   public void onSuccess(CodeUrl codeUrl) {
143                     GWT.log("Code url: " + codeUrl.getUrl());
144                     display.setCodeUrl(
145                         codeUrl.getClassName(),
146                         codeUrl.getUrl());
147                   }
148 
149                   @Override
150                   public void onFailure(Throwable caught) {
151                     GWT.log("not able to retreive code due to: ",
152                         caught);
153                     display.setCode("no code available");
154                   }
155                 });
156 
157           }
158         });
159   }
