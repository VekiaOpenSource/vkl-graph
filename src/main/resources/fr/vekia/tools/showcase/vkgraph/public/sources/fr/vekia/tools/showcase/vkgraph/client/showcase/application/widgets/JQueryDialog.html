072         Scheduler.get().scheduleDeferred(new ScheduledCommand() {
073 
074             @Override
075             public void execute() {
076                 if (!sized) {
077                     if (autoSize) {
078                         height = scrollableContent.getOffsetHeight() + HEADER_HEIGHT;
079                         width = scrollableContent.getOffsetWidth();
080                     } else {

083                     }
084                 }
085                 open(id, height, width);
086                 isShowing = true;
087                 RootLayoutPanel.get().remove(JQueryDialog.this);
088 
089                 if (javaCode != null) {
090                     // remove header height + scroll bar bottom
091                     buildCodeMirror(codeElementId, javaCode, height - HEADER_HEIGHT - 20);
092                 }
093             }
094         });
095     };
