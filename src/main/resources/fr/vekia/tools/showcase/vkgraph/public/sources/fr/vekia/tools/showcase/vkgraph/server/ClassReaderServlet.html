49     /** Read the contents of the given file. */
50     private StringBuilder read(InputStream input) throws IOException {
51         StringBuilder text = new StringBuilder();
52         String lineSeparator = System.getProperty("line.separator");
53         Scanner scanner = new Scanner(input, "UTF-8");
54         try {
55             while (scanner.hasNextLine()) {
56                 text.append(scanner.nextLine() + lineSeparator);
57             }
58         } finally {
59             scanner.close();
60         }
61         return text;
62     }
