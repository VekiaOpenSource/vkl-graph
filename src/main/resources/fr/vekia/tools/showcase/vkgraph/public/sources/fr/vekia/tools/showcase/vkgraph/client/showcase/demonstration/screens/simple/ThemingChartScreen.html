152     private IsWidget getFlexTableThemeCreator() {
153         FlexTable table = new FlexTable();
154         table.setStyleName("gwt-DataTable");
155         table.setSize("100%", "100%");
156 
157         final List<TextBox> colors = new ArrayList<TextBox>();
158 
159         final TextBox themeNameBox = new TextBox();
160         themeNameBox.setText(DOM.createUniqueId());
161         int row = 0;
162 
163         table.setText(0, 0, "theme name");
164         table.setWidget(0, 1, themeNameBox);
165         table.getRowFormatter().setStyleName(row, "gwt-DataTableTitle");
166         row++;
167         table.setText(1, 0, "Property");
168         table.setText(1, 1, "Value");
169         table.getRowFormatter().setStyleName(row, "gwt-DataTableTitle");
170         row++;
171         final TextBox backgroundGridColorBox = new TextBox();
172         table.setText(row, 0, "Background Grid");
173         table.setWidget(row, 1, backgroundGridColorBox);
174         backgroundGridColorBox.addValueChangeHandler(new ValueChangeHandler<String>() {
175 
176             @Override
177             public void onValueChange(ValueChangeEvent<String> event) {
178                 backgroundGridColorBox.getElement().getStyle().setBackgroundColor(backgroundGridColorBox.getText());
179             }
180         });
181 
182         row++;
183 
184         for (int i = 0; i < 10; i++) {
185             final TextBox seriesColor = new TextBox();
186             table.setText(row, 0, "Serie " + (i + 1) + "color");
187             table.setWidget(row, 1, seriesColor);
188             colors.add(seriesColor);
189 
190             seriesColor.addValueChangeHandler(new ValueChangeHandler<String>() {
191 
192                 @Override
193                 public void onValueChange(ValueChangeEvent<String> event) {
194                     seriesColor.getElement().getStyle().setBackgroundColor(seriesColor.getText());
195                 }
196             });
197             row++;
198         }
199         Button generate = new Button("Validate");
200         generate.addClickHandler(new ClickHandler() {
201 
202             @Override
203             public void onClick(ClickEvent event) {
204                 JSONObject grid = new JSONObject();
205                 JSONObject color = new JSONObject();
206                 color.put("backgroundColor", new JSONString(backgroundGridColorBox.getText()));
207                 grid.put("grid", color);
208 
209                 JSONArray colorsArray = new JSONArray();
210 
211                 int i = 0;
212                 for (TextBox widget : colors) {
213                     if (!widget.getText().isEmpty()) {
214                         JSONObject colorSerie = new JSONObject();
215                         colorSerie.put("color", new JSONString(widget.getText()));
216                         colorsArray.set(i, colorSerie);
217                         i += 1;
218                     }
219                 }
220                 grid.put("series", colorsArray);
221 
222                 first.activateTheme(themeNameBox.getText(), grid.getJavaScriptObject());
223             }
224         });
225         row++;
226 
227         CheckBox checkBox = new CheckBox("Enable change default theme by right click");
228         checkBox.addValueChangeHandler(new ValueChangeHandler<Boolean>() {
229 
230             @Override
231             public void onValueChange(ValueChangeEvent<Boolean> event) {
232                 first.setThemingEnable(event.getValue());
233                 second.setThemingEnable(event.getValue());
234                 third.setThemingEnable(event.getValue());
235             }
236         });
237         table.setWidget(row, 0, checkBox);
238 
239         table.setWidget(row, 1, generate);
240 
241         return table;
242     }
