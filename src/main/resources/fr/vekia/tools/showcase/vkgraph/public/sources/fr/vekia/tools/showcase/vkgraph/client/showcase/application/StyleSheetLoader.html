001 package fr.vekia.tools.showcase.vkgraph.client.showcase.application;
002 
003 import com.google.gwt.core.client.Scheduler;
004 import com.google.gwt.dom.client.Document;
005 import com.google.gwt.dom.client.HeadElement;
006 import com.google.gwt.dom.client.LinkElement;
007 import com.google.gwt.user.client.Command;
008 import com.google.gwt.user.client.Timer;
009 import com.google.gwt.user.client.ui.Label;
010 import com.google.gwt.user.client.ui.RootLayoutPanel;
011 
012 /**
013  * An utility class that loads style sheets.
014  * 
015  * @author svandecappelle
016  * @since Dec 14, 2012. VklGraph version 1.2
017  * @version 2.1
018  */
019 public class StyleSheetLoader {
020     /**
021      * A {@link Timer} that creates a small reference widget used to determine when a new style sheet has finished
022      * loading. The widget has a natural width of 0px, but when the style sheet is loaded, the width changes to 5px. The
023      * style sheet should contain a style definition that is passed into the constructor that defines a height and width
024      * greater than 0px.
025      */
026     private static class StyleTesterTimer extends Timer {
027         private Command callback;
028         private Label refWidget;
029 
030         /**
031          * Create a new {@link StyleTesterTimer}.
032          * 
033          * @param refStyleName
034          *            the reference style name
035          * @param callback
036          *            the callback to execute when the style sheet loads
037          */
038         public StyleTesterTimer(String refStyleName, Command callback) {
039             this.callback = callback;
040 
041             // Create the reference Widget
042             refWidget = new Label();
043             refWidget.setStyleName(refStyleName);
044             refWidget.getElement().getStyle().setProperty("position", "absolute");
045             refWidget.getElement().getStyle().setProperty("visibility", "hidden");
046             refWidget.getElement().getStyle().setProperty("display", "inline");
047             refWidget.getElement().getStyle().setPropertyPx("padding", 0);
048             refWidget.getElement().getStyle().setPropertyPx("margin", 0);
049             refWidget.getElement().getStyle().setPropertyPx("border", 0);
050             refWidget.getElement().getStyle().setPropertyPx("top", 0);
051             refWidget.getElement().getStyle().setPropertyPx("left", 0);
052             RootLayoutPanel.get().add(refWidget);
053         }
054 
055         @Override
056         public void run() {
057             // Redisplay the reference widget so it redraws itself
058             refWidget.setVisible(false);
059             refWidget.setVisible(true);
060 
061             // Check the dimensions of the reference widget
062             if (refWidget.getOffsetWidth() > 0) {
063                 RootLayoutPanel.get().remove(refWidget);
064 
065                 // Fire the callback in a DeferredCommand to ensure the browser
066                 // has
067                 // enough time to parse the styles. Otherwise, we'll get weird
068                 // styling
069                 // issues.
070                 Scheduler.get().scheduleDeferred(callback);
071             } else {
072                 schedule(10);
073             }
074         }
075     }
076 
077     /**
078      * Convenience method for getting the document's head element.
079      * 
080      * @return the document's head element
081      */
082     public static native HeadElement getHeadElement()
083     /*-{
084       return $doc.getElementsByTagName("head")[0];
085     }-*/;
086 
087     /**
088      * Load a style sheet onto the page.
089      * 
090      * @param href
091      *            the url of the style sheet
092      */
093     public static void loadStyleSheet(String href) {
094         LinkElement linkElem = Document.get().createLinkElement();
095         linkElem.setRel("stylesheet");
096         linkElem.setType("text/css");
097         linkElem.setHref(href);
098         getHeadElement().appendChild(linkElem);
099     }
100 
101     /**
102      * Load a style sheet onto the page and fire a callback when it has loaded. The style sheet should contain a style
103      * definition called refStyleName that defines a height and width greater than 0px.
104      * 
105      * @param href
106      *            the url of the style sheet
107      * @param refStyleName
108      *            the style name of the reference element
109      * @param callback
110      *            the callback executed when the style sheet has loaded
111      */
112     public static void loadStyleSheet(String href, String refStyleName, Command callback) {
113         loadStyleSheet(href);
114         waitForStyleSheet(refStyleName, callback);
115     }
116 
117     /**
118      * Detect when a style sheet has loaded by placing an element on the page that is affected by a rule in the style
119      * sheet, as described in {@link #loadStyleSheet(String, String, Command)}. When the style sheet has loaded, the
120      * callback will be executed.
121      * 
122      * @param refStyleName
123      *            the style name of the reference element
124      * @param callback
125      *            the callback executed when the style sheet has loaded
126      */
127     public static void waitForStyleSheet(String refStyleName, Command callback) {
128         new StyleTesterTimer(refStyleName, callback).run();
129     }
130 }
