019     public static void animateShow(Element element, AnimationEnum animation, int duration) {
020         if (element.getId().isEmpty()) {
021             element.setId(DOM.createUniqueId());
022         }
023         switch (animation) {
024         case EXPLODE:
025             explode("#" + element.getId(), true, duration);
026             break;
027         case FADE_IN:
028             fadeIn("#" + element.getId(), duration);
029             break;
030         case FADE_OUT:
031             fadeOut("#" + element.getId(), duration);
032             break;
033         case FADE_TOGGLE:
034             fadeToggle("#" + element.getId(), duration);
035             break;
036         case SLIDE_DOWN:
037             slideDown("#" + element.getId(), true, duration);
038             break;
039         case SLIDE_TOGGLE:
040             slideToggle("#" + element.getId(), true, duration);
041             break;
042         case SLIDE_UP:
043             slideUp("#" + element.getId(), true, duration);
044             break;
045         default:
046             break;
047         }
048     }
