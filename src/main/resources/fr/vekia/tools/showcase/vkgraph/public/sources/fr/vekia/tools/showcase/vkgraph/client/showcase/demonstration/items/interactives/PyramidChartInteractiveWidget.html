040   public PyramidChartInteractiveWidget() {
041 
042     final PyramidChart chart = getChart();
043     final FlexTable datasTable = new FlexTable();
044     datasTable.setText(0, 0, "Sex");
045     datasTable.setText(1, 0, "Age");
046     datasTable.setText(2, 0, "Ratio");
047 
048     chart.bind(new DataHighlightEvent() {
049 
050       @Override
051       public void onEvent(EventObject datasOnEvent) {
052         if (datasOnEvent.getSeriesIndex() == 0) {
053           datasTable.setText(0, 1, "Male");
054         } else {
055           datasTable.setText(0, 1, "Female");
056         }
057 
058         datasTable.setText(1, 1,
059             Integer.toString(datasOnEvent.getPointIndex() + 1));
060         String ratio = datasOnEvent.getValue().toString()
061             .replaceFirst("([0-9])*,", "");
062         datasTable.setText(2, 1, ratio);
063       }
064     });
065 
066     add(datasTable);
067 
068     datasTable.addStyleName("gwt-DataTable");
069     datasTable.getRowFormatter().setStyleName(0, "gwt-DataTableTitle");
070     datasTable.setWidth("100%");
071 
072     add(chart);
073     add(datasTable);
074     setCellWidth(datasTable, "100%");
075     chart.setResizable(true);
076   }
