216   /**
217    * @param plots
218    */
219   private void putChildrens(AbstractShowcaseTreeMenuItem plotItem) {
220     for (AbstractShowcaseTreeMenuItem treeItem : plotItem
221         .getChildrenItems()) {
222       plotsScreens.put(treeItem.getText(), treeItem);
223       if (treeItem.hasChildren()) {
224         putChildrens(treeItem);
225       }
226     }

241   private void selectCategoryAnimation(
242       final SelectionOnMenuRequiredEvent event) {
243     Timer mouseCategorySelection = new Timer() {
244 
245       @Override
246       public void run() {
247         ((Display) display).showWidget(categories.get(event
248             .getMenuToShow()));
249         ((Display) display).addHeaderStyleName("AutomatedDemo-Focus",
250             categories.get(event.getMenuToShow()));
251         if (event.getPlotToShow() != null) {
252           selectOnTreeAnimation(event);
253         }
254       }
255     };
256     UIObject headerWidget = ((Display) display).getHeaderWidget(categories
257         .get(event.getMenuToShow()));
258     MouseDemoWidget mouse = MouseDemoWidget.getInstance();
259     mouse.scrollTo(headerWidget, 1000);
260     mouseCategorySelection.schedule(1300);
261   }

279   private void selectOnTreeAnimation(final SelectionOnMenuRequiredEvent event) {
280     final MouseDemoWidget mouse = MouseDemoWidget.getInstance();
281     final Timer hideSelectionTree = new Timer() {
282 
283       @Override
284       public void run() {
285         ((Display) display).removeItemStyleName(event.getMenuToShow(),
286             event.getPlotToShow(), "AutomatedDemo-Focus");
287       }
288     };
289 
290     final Timer selectionTree = new Timer() {
291 
292       @Override
293       public void run() {
294         ((Display) display).selectTreeItem(event.getMenuToShow(),
295             plotsScreens.get(event.getPlotToShow()));
296         ShowcaseInjector.Util.getInstance().getConsoleCodePresenter()
297             .showConsole();
298         ((Display) display).removeHeaderStyleName(
299             "AutomatedDemo-Focus",
300             categories.get(event.getMenuToShow()));
301         ((Display) display).addItemStyleName(event.getMenuToShow(),
302             event.getPlotToShow(), "AutomatedDemo-Focus");
303         hideSelectionTree.schedule(3500);
304       }
305     };
306 
307     Timer selectiontreeMouse = new Timer() {
308 
309       @Override
310       public void run() {
311         UIObject treeItem = ((Display) display).getTreeItem(
312             event.getMenuToShow(), event.getPlotToShow());
313         mouse.scrollTo(treeItem, 1000);
314         selectionTree.schedule(1300);
315       }
316     };
317     selectiontreeMouse.schedule(500);
318   }
