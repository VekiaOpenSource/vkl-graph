047     private static String CUR_THEME;
048 
049     public static void setTheme(Theme theme) {
050         CUR_THEME = theme.getThemeName();
051         updateStyleSheets();
052     }

054     /**
055      * Update the style sheets to reflect the current theme and direction.
056      */
057     private static void updateStyleSheets() {
058         // Generate the names of the style sheets to include
059         String gwtStyleSheet = "css/themes/" + CUR_THEME + "/" + CUR_THEME + ".css";
060         if (LocaleInfo.getCurrentLocale().isRTL()) {
061             gwtStyleSheet = gwtStyleSheet.replace(".css", "_rtl.css");
062         }
063 
064         // Find existing style sheets that need to be removed
065         boolean styleSheetsFound = false;
066         final HeadElement headElem = StyleSheetLoader.getHeadElement();
067         final List<Element> toRemove = new ArrayList<Element>();
068         NodeList<Node> children = headElem.getChildNodes();
069         for (int i = 0; i < children.getLength(); i++) {
070             Node node = children.getItem(i);
071             if (node.getNodeType() == Node.ELEMENT_NODE) {
072                 Element elem = Element.as(node);
073                 if (elem.getTagName().equalsIgnoreCase("link") && elem.getPropertyString("rel").equalsIgnoreCase("stylesheet")) {
074                     styleSheetsFound = true;
075                     String href = elem.getPropertyString("href");
076                     // If the correct style sheets are already loaded, then we
077                     // should have
078                     // nothing to remove.
079                     if (!href.contains(gwtStyleSheet) && notAdependencyTheme(href)) {
080                         // toRemove.add(elem);
081                     }
082                 }
083             }
084         }
085 
086         // Return if we already have the correct style sheets
087         if (styleSheetsFound && toRemove.size() == 0) {
088             return;
089         }
090 
091         // Detach the app while we manipulate the styles to avoid rendering
092         // issues
093 
094         // Remove the old style sheets
095         for (Element elem : toRemove) {
096             headElem.removeChild(elem);
097         }
098 
099         // Load the GWT theme style sheet
100         Command callback = new Command() {
101             /**
102              * The number of style sheets that have been loaded and executed this command.
103              */
104             private int numStyleSheetsLoaded = 0;
105 
106             public void execute() {
107                 // Wait until all style sheets have loaded before re-attaching
108                 // the app
109                 numStyleSheetsLoaded++;
110                 if (numStyleSheetsLoaded < 2) {
111                     return;
112                 }
113 
114                 // Different themes use different background colors for the body
115                 // element, but IE only changes the background of the visible
116                 // content
117                 // on the page instead of changing the background color of the
118                 // entire
119                 // page. By changing the display style on the body element, we
120                 // force
121                 // IE to redraw the background correctly.
122                 RootPanel.getBodyElement().getStyle().setProperty("display", "none");
123                 RootPanel.getBodyElement().getStyle().setProperty("display", "");
124                 ShowcaseInjector injector = ShowcaseInjector.Util.getInstance();
125                 ShowcasePresenter presenter = injector.getPresenter();
126                 presenter.revealDisplay();
127                 // ThemeInjector injectorTheme =
128                 // ThemeInjector.Util.getInstance();
129                 // injectorTheme.getPresenter().revealDisplay();
130             }
131         };
132         StyleSheetLoader.loadStyleSheet(gwtStyleSheet, getCurrentReferenceStyleName("gwt"), callback);
133 
134         // Load the showcase specific style sheet after the GWT theme style
135         // sheet so
136         // that custom styles supercede the theme styles.
137         // StyleSheetLoader.loadStyleSheet(showcaseStyleSheet, getCurrentReferenceStyleName("Application"), callback);
138     }

140     /**
141      * @param href
142      * @return
143      */
144     private static boolean notAdependencyTheme(String href) {
145         boolean isNotADependency = true;
146         if (href.contains("jquery-ui-1.8.23.custom.css")) {
147             isNotADependency = false;
148         } else if (href.contains("jquery.jqplot.css")) {
149             isNotADependency = false;
150         } else if (href.contains("gwt/clean/clean.css")) {
151             isNotADependency = false;
152         }
153 
154         return isNotADependency;
155     }

157     /**
158      * Get the style name of the reference element defined in the current GWT theme style sheet.
159      * 
160      * @param prefix
161      *            the prefix of the reference style name
162      * @return the style name
163      */
164     private static String getCurrentReferenceStyleName(String prefix) {
165         String gwtRef = prefix + "-Reference-" + CUR_THEME;
166         if (LocaleInfo.getCurrentLocale().isRTL()) {
167             gwtRef += "-rtl";
168         }
169         return gwtRef;
170     }
