021 public class ConsoleCode implements Display {
022 
023     private Button buttonShow;
024     private JQueryDialog box;
025     private String codeUrl;
026     private String className;
027     private Anchor showOtherWindow;
028     private FlowPanel container;
029 
030     /**
031      * Default constructor
032      * 
033      */
034     public ConsoleCode() {
035         this.container = new FlowPanel();
036         box = new JQueryDialog("The code", false);
037         buttonShow = new Button("ShowCode");
038         showOtherWindow = new Anchor("Show in other window");
039         showOtherWindow.setTarget("_blank");
040         buttonShow.getElement().setId("idOpener");
041         container.add(buttonShow);
042         container.add(showOtherWindow);
043         container.setStylePrimaryName("Vklgetcode-Container");
044     }
045 
046     /**
047      * @return
048      */
049     protected String getCodeUrl() {
050         return this.codeUrl;
051     }
052 
053     /**
054      * @return
055      */
056     @Override
057     public void setCodeUrl(String className, String codeUrl) {
058         this.codeUrl = codeUrl;
059         this.className = className;
060         this.showOtherWindow.setHref(GWT.getHostPageBaseURL() + getCodeUrl());
061     }
062 
063     /**
064      * @return the className
065      */
066     public String getClassName() {
067         return className;
068     }
069 
070     @Override
071     public Widget asWidget() {
072         return buttonShow;
073     }
074 
075     @Override
076     public void show() {
077         box.show();
078     }
079 
080     @Override
081     public void hide() {
082         // box.hide();
083     }
084 
085     @Override
086     public HasClickHandlers getAddShowHandler() {
087         return buttonShow;
088     }
089 
090     @Override
091     public HasClickHandlers getAddHideHandler() {
092         return null;
093     }
094 
095     @Override
096     public void setCode(String code) {
097         box.setCode(code);
098     }
099 
100     @Override
101     public void insert() {
102         RootLayoutPanel.get().add(container);
103         container.getElement().getStyle().setZIndex(1);
104         RootLayoutPanel.get().setWidgetTopHeight(container, 16, Unit.PX, 45, Unit.PX);
105         RootLayoutPanel.get().setWidgetRightWidth(container, 16, Unit.PX, 260, Unit.PX);
106     }
107 
108     @Override
109     public void showOutBrowser() {
110         Window.open(GWT.getHostPageBaseURL() + getCodeUrl(), "Code for " + getClassName(), "_blank");
111     }
112 }
