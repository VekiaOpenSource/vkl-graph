036 public class DraggingPointExempleScreen extends SimplePanel {
037     private LineChart<List<Number>> chart;
038 
039     /**
040      * Default constructor
041      * 
042      */
043     public DraggingPointExempleScreen() {
044 
045         VerticalPanel panel = new VerticalPanel();
046 
047         chart = new LineChart<List<Number>>();
048         chart.setOption(ChartOption.title, "tutu");
049         ArrayList<List<Number>> data = new ArrayList<List<Number>>();
050         ArrayList<Number> dataLine1 = new ArrayList<Number>();
051         ArrayList<Number> dataLine2 = new ArrayList<Number>();
052 
053         for (int i = 0; i < 10; i++) {
054             dataLine1.add(Random.nextInt(250));
055             dataLine2.add(dataLine1.get(i));
056         }
057 
058         chart.setSize("1200px", "660px");
059 
060         data.add(dataLine1);
061         data.add(dataLine2);
062 
063         chart.setDatas(data);
064         chart.setExportEnable(true);
065         chart.setRefreshManuallyEnable(true);
066         chart.setTextOption(ChartOption.grid, SubOption.background, "white");
067         chart.setBooleanOption(ChartOption.cursor, SubOption.show, true);
068 
069         SeriesData dataserie = new SeriesData();
070         OptionSerie serie1 = new OptionSerie();
071         OptionSerie serie2 = new OptionSerie();
072 
073         serie1.setBooleanOption(SubOption.isDragable, false);
074         serie1.setBooleanOption(SubOption.trendline, SubOption.show, false);
075         serie1.setBooleanOption(SubOption.showMarker, false);
076 
077         serie2.setTextOption(SubOption.dragable, SubOption.constrainTo, "y");
078         serie2.setBooleanOption(SubOption.trendline, SubOption.show, true);
079 
080         dataserie.set(0, serie1);
081         dataserie.set(1, serie2);
082 
083         List<String> colors = new ArrayList<String>();
084         colors.add("rgba(255, 0, 0,0.4)");
085         colors.add("rgba(78, 135, 194, 0.7)");
086 
087         chart.setColors(colors);
088         chart.setPluginsEnable(true);
089         chart.setSeriesOptions(dataserie);
090 
091         chart.bind(new EditedDataEvent() {
092 
093             @Override
094             public void onEvent(EventObject datasOnEvent) {
095                 GWT.log("edited value: [" + datasOnEvent.getSeriesIndex() + "-" + datasOnEvent.getPointIndex() + "] - {" + datasOnEvent.getValue() + "}");
096             }
097         });
098 
099         HorizontalPanel editorPanel = new HorizontalPanel();
100         editorPanel.setVerticalAlignment(HasVerticalAlignment.ALIGN_MIDDLE);
101         editorPanel.add(new InlineLabel("Background color: "));
102 
103         final TextBox gridAttribute = new TextBox();
104         gridAttribute.setText("white");
105 
106         final Button replotCheckBox = new Button("New random");
107 
108         chart.addToRightClickMenu(new MenuCommands("Custom command") {
109 
110             @Override
111             public void execute() {
112                 Window.alert("A custom command fired");
113             }
114         });
115         chart.setThemingEnable(true);
116         replotCheckBox.addClickHandler(new ClickHandler() {
117 
118             @Override
119             public void onClick(ClickEvent event) {
120                 List<List<Number>> dataReploted = new ArrayList<List<Number>>();
121 
122                 List<Number> dataLineIdx1 = new ArrayList<Number>();
123                 List<Number> dataLineIdx2 = new ArrayList<Number>();
124 
125                 for (int i = 0; i < 10; i++) {
126                     dataLineIdx1.add(Random.nextInt(250));
127                     dataLineIdx2.add(Random.nextInt(250));
128                 }
129                 dataReploted.add(dataLineIdx1);
130                 dataReploted.add(dataLineIdx2);
131 
132                 DraggingPointExempleScreen.this.chart.setDatas(dataReploted);
133                 DraggingPointExempleScreen.this.chart.replot();
134             }
135         });
136 
137         editorPanel.add(gridAttribute);
138         editorPanel.add(replotCheckBox);
139 
140         gridAttribute.addValueChangeHandler(new ValueChangeHandler<String>() {
141 
142             @Override
143             public void onValueChange(ValueChangeEvent<String> event) {
144 
145                 DraggingPointExempleScreen.this.chart.setTextOption(ChartOption.series, SubOption.color, gridAttribute.getText());
146                 DraggingPointExempleScreen.this.chart.replot();
147             }
148         });
149 
150         panel.add(new HTML("<h2>Try to right clic on the chart. And Change the theme on the available selection.</h2>"));
151         panel.add(chart);
152         panel.add(editorPanel);
153         setWidget(panel);
154     }
155 
156 }
