052   private int positionPercentTop;

099   /**
100    * @param x
101    * @param y
102    * @param milliseconds
103    */
104   public void scrollTo(int x, int y, int milliseconds) {
105     this.finalX = x;
106     this.finalY = y;
107 
108     startX = messageBox.getElement().getOffsetLeft();
109     startY = messageBox.getElement().getOffsetTop();
110 
111     run(milliseconds);
112   }
113 
114   @Override
115   protected void onUpdate(double progress) {
116     double positionX = startX + (progress * (this.finalX - startX));
117     double positionY = startY + (progress * (this.finalY - startY));
118 
119     this.messageBox.getElement().getStyle()
120         .setLeft(positionX, Style.Unit.PX);
121     this.messageBox.getElement().getStyle()
122         .setTop(positionY, Style.Unit.PX);
123   }
124 
125   @Override
126   protected void onComplete() {
127     super.onComplete();
128     this.messageBox.getElement().getStyle()
129         .setLeft(this.finalX, Style.Unit.PX);
130     this.messageBox.getElement().getStyle()
131         .setTop(this.finalY, Style.Unit.PX);
132   }
133 
134   /**
135      * 
136      */
137   public void scrollCenter(final int milliseconds) {
138     final Number positionFinalX = Window.getClientWidth()
139         * (DEFAULT_WIDGET_ANIMATION_X_END_POSITION / 100f)
140         - (messageBox.getOffsetHeight() / 2);
141     final Number positionFinalY = Window.getClientHeight()
142         * (DEFAULT_WIDGET_ANIMATION_Y_END_POSITION / 100f)
143         - (messageBox.getOffsetWidth() / 2);
144     scrollTo(positionFinalX.intValue(), positionFinalY.intValue(),
145         milliseconds / 2);
146   }
147 
148   private void show(AnimationEnum animation) {
149     messageBox.show();
150     if (animation == AnimationEnum.SCROLLCENTER_BOUNCE) {
151       SequentialTimerStack timers = new SequentialTimerStack();
152       timers.addToStack(new SequentialTimer(10) {
153 
154         @Override
155         public void execute() {
156           scrollCenter(MILLESECONDS_DEFAULT_EVENT_DURATION);
157         }
158       });
159       timers.addToStack(new SequentialTimer(400) {
160 
161         @Override
162         public void execute() {
163           animate(AnimationEnum.BOUNCE, 3,
164               MILLESECONDS_FAST_EVENT_DURATION);
165         }
166       });
167 
168       timers.run();
169     } else {
170       JqueryAnimation.animateShow(messageBox.getElement(), animation,
171           MILLESECONDS_SLOW_EVENT_DURATION);
172     }
173   }
174 
175   private void show(UIObject objectRelativeToShow, AnimationEnum animation) {
176     if (!notYetSet || resetPositionPopupToEachVisible) {
177       messageBox.showRelativeTo(objectRelativeToShow);
178       notYetSet = true;
179     } else {
180       messageBox.show();
181     }
182     if (animation != null) {
183       JqueryAnimation.animateShow(messageBox.getElement(), animation,
184           MILLESECONDS_SLOW_EVENT_DURATION);
185     }
186   }

197   public void setPopupAnimationStartPosition(int startxPosition, Unit unitx,

199     this.unitx = unitx;
200     this.unity = unity;
201 
202     this.positionPercentLeft = startxPosition;
203     this.positionPercentTop = startyPosition;
204     resetStartPosition();
205   }
206 
207   private void resetStartPosition() {
208     Number startx;
209     Number starty;
210     if (unitx == Unit.PCT) {
211       startx = Window.getClientWidth() * (positionPercentLeft / 100f)
212           - (messageBox.getOffsetHeight() / 2);
213     } else {
214       startx = positionPercentLeft;
215     }
216     if (unity == Unit.PCT) {
217       starty = Window.getClientHeight() * (positionPercentTop / 100f)
218           - (messageBox.getOffsetWidth() / 2);
219     } else {
220       starty = positionPercentTop;
221     }
222 
223     messageBox.setPopupPosition(startx.intValue(), starty.intValue());
224   }

226   public void hide() {
227     if (animationHide != null) {
228       hide(animationHide);
229     } else {
230       messageBox.hide();
231     }
232   }

234   private void hide(AnimationEnum animation) {
235     JqueryAnimation.animateHide(messageBox.getElement(), animation,
236         MILLESECONDS_SLOW_EVENT_DURATION);
237     SequentialTimerStack timers = new SequentialTimerStack();
238     timers.addToStack(new SequentialTimer(MILLESECONDS_SLOW_EVENT_DURATION
239         + MILLESECONDS_FAST_EVENT_DURATION) {
240 
241       @Override
242       public void execute() {
243         messageBox.hide();
244       }
245     });
246     timers.run();
247   }
