285                         SubOption subOption = SubOption.valueOf(ListSubOption.this.getValue(ListSubOption.this.getSelectedIndex()));

303         private ChartOption getOptionChart() {

305             List<SubOption> subOptions = new ArrayList<SubOption>();
306

308                 if (chartHasIsOption.getOption() instanceof SubOption) {
309                     subOptions.add((SubOption) chartHasIsOption.getOption());
310                 }
311                 chartHasIsOption = chartHasIsOption.getParentOption();
312             }
313 
314             // option Chart

317         }

360     private void changeWidget(final int rowId, final int columnId, final HasType type, final ChartOption optionChart, SubOption subOption, final List<SubOption> allSubOptions, HasIsOption hasIsOption) {
361         Widget widget = null;
362         if (subOption != null && type.getType() == null) {
363             widget = new ListSubOption(rowId, columnId, subOption, hasIsOption);
364         } else if (type.getType() == OptionType.BOOLEAN) {
365             if (type.hasChildren()) {
366                 ListSubOption listOptions = new ListSubOption(rowId, 1, hasIsOption.getOption(), hasIsOption);
367                 listOptions.addItem("", "NONE");
368 
369                 CheckBox checkBox = new CheckBox();
370                 checkBox.addValueChangeHandler(new ValueChangeHandler<Boolean>() {
371 
372                     @Override
373                     public void onValueChange(ValueChangeEvent<Boolean> event) {
374                         apply(optionChart, allSubOptions, event.getValue());
375                     }
376 
377                 });
378 
379                 optionsTable.setWidget(rowId, columnId + 1, checkBox);
380 
381                 widget = listOptions;
382             } else {
383 
384                 CheckBox checkBox = new CheckBox();
385                 checkBox.addValueChangeHandler(new ValueChangeHandler<Boolean>() {
386 
387                     @Override
388                     public void onValueChange(ValueChangeEvent<Boolean> event) {
389                         apply(optionChart, allSubOptions, event.getValue());
390                     }
391 
392                 });
393                 widget = checkBox;
394             }
395 
396         } else if (type.getType() == OptionType.NUMBER) {
397             TextBox box = new TextBox();
398             box.addValueChangeHandler(new ValueChangeHandler<String>() {
399 
400                 @Override
401                 public void onValueChange(ValueChangeEvent<String> event) {
402                     apply(optionChart, allSubOptions, Float.parseFloat(event.getValue()));
403                 }
404             });
405             widget = box;
406         } else if (type.getType() == OptionType.STRING) {
407             TextBox box = new TextBox();
408             box.addValueChangeHandler(new ValueChangeHandler<String>() {
409 
410                 @Override
411                 public void onValueChange(ValueChangeEvent<String> event) {
412                     apply(optionChart, allSubOptions, event.getValue());
413                 }
414             });
415             widget = box;
416         }
417         if (widget != null) {
418             optionsTable.setWidget(rowId, columnId, widget);
419             if (!(widget instanceof ListSubOption) && optionsTable.getCellCount(rowId) > columnId + 1) {
420                 optionsTable.removeCells(rowId, columnId + 1, optionsTable.getCellCount(rowId) - (columnId + 1));
421             }
422         }
423     }
