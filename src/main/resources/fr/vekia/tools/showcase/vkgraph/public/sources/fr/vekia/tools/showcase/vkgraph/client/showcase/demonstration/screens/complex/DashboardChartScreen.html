026 public class DashboardChartScreen extends DashboardChart<List<DualValue>> {
027 
028     /**
029      * Default constructor
030      * 
031      */
032     public DashboardChartScreen() {
033 
034         // Datas chart
035         List<List<DualValue>> dataLines = new ArrayList<List<DualValue>>();
036 
037         // Date xaxis
038         List<Date> dates = new ArrayList<Date>();
039         for (int i = 0; i < 50; i++) {
040             Date date = new Date();
041             CalendarUtil.addDaysToDate(date, i);
042             dates.add(date);
043         }
044 
045         // #############
046         // ###Â Datas ###
047         // #############
048 
049         List<DualValue> data1 = new ArrayList<DualValue>();
050 
051         // line 1
052         for (Date dateXaxis : dates) {
053             data1.add(new DualValue(dateXaxis.getTime(), Random.nextInt(10) + 1));
054         }
055         // line 2
056         List<DualValue> data2 = new ArrayList<DualValue>();
057         for (DualValue dualValue : data1) {
058             data2.add(new DualValue(dualValue.getKeyNumber(), dualValue.getValue().intValue() + 1 + Random.nextInt(10)));
059         }
060 
061         dataLines.add(data1);
062         dataLines.add(data2);
063 
064         // Graph instances
065         super.setData(dataLines);
066         super.setTitle("Exemple of dashboard");
067         super.setBackgroundColor("rgba(57,57,57,1.0)");
068 
069         // ##############
070         // ###Â series ###
071         // ##############
072         SeriesData series = new SeriesData();
073 
074         OptionSerie serie1 = new OptionSerie();
075         serie1.setBooleanOption(SubOption.fill, true);
076         serie1.setTextOption(SubOption.label, "Serie 1");
077 
078         OptionSerie serie2 = new OptionSerie();
079         serie2.setTextOption(SubOption.label, "Serie 2");
080 
081         series.set(0, serie1);
082         series.set(1, serie2);
083         super.setSeries(series);
084 
085         super.setXFormat("%b %e");
086         super.setYFormat("$%'d");
087 
088         // xaxis
089         super.setXaxisRenderer(RenderersEnum.DateAxis);
090         // yaxis
091         super.setYaxisRenderer(RenderersEnum.LogAxis);
092 
093         super.drawXGrid(true);
094         super.drawYGrid(true);
095         super.setExportEnable(true);
096 
097         CanvasOverlayObject objectsCanvas = new CanvasOverlayObject();
098         Rectangle rect = new Rectangle();
099         rect.setTextOption(SubOption.xmin, "2014-06-27 05:00:00");
100         rect.setTextOption(SubOption.xmax, "2014-07-07 05:00:00");
101 
102         rect.setNumberOption(SubOption.ymin, 2);
103         rect.setNumberOption(SubOption.ymax, 15);
104 
105         rect.setTextOption(SubOption.color, "rgb(100, 55, 124)");
106         rect.setBooleanOption(SubOption.showTooltip, true);
107         rect.setTextOption(SubOption.tooltipFormatString, "<b><i><span style='color:red;''>Test2</span></i></b>");
108         rect.setTextOption(SubOption.tooltipLocation, "s");
109         rect.setTextOption(SubOption.xformat, SubOption.type, "date");
110         rect.setTextOption(SubOption.xformat, SubOption.format, "%Y-%m-%d %H:%M:%S");
111 
112         objectsCanvas.addObject(rect);
113         super.setCanvasOverlay(objectsCanvas);
114     }
